class Solution {
public:
    vector<int> twoSum(vector<int>& numbers, int target) {
       
        unordered_map<int, int> hash_table;

       
        for (int i = 0; i < numbers.size(); i++) {
            int current_num = numbers[i];
            int complement = target - current_num;

           // Check if the complement exists in our map
            if (hash_table.find(complement) != hash_table.end()) {
                // The problem requires 1-based indexing
                // hash_table[complement] + 1 is the first index
                // i + 1 is the current index
                return {hash_table[complement] + 1, i + 1};
            }

            // If not found, storing current number and its index
            hash_table[current_num] = i;
        }

       
        return {};
    }
};
